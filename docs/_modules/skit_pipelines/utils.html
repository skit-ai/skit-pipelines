[docs]    def ping(self, cc):
        if not cc:
            return self

        names = []
        for name in cc.split(","):
            name = name.strip()
            if not name.startswith("<@"):
                name = f"<@{name}>"
            if not name:
                continue
            names.append(name)
        cc_group = " ".join(names)
        self.content = f"{self.content}\ncc: {cc_group}"
        return self
