[docs]def upload2tog(input_file: str, token: str, job_ids: List[str], data_label: str):
    url = pipeline_constants.CONSOLE_API_URL
    errors = []
    df_sizes = []
    for job_id in job_ids:
        error, df_size = upload_dataset(
            input_file, url, token, job_id, labels_constants.SOURCE__DB, data_label
        )
        if error:
            errors.extend(error)
        df_sizes.append(df_size)
    return errors, df_sizes
